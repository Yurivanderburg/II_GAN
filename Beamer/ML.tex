    \begin{frame}{Requirements of High Resolution}
    	\begin{columns}[c]
    		\column{0.6\textwidth}
    		\begin{itemize}
    			\item Stellar objects appear almost like point sources.
    			\item Angular sizes are extremely small ($\sim$ milli-arcsecond or less).
    			\item High angular resolution is required to resolve stellar surfaces.
    			\item Understanding stellar structure is crucial for stellar evolution.
    			\item The Sun is the only star currently resolved in detail.
    		\end{itemize}

    \begin{frame}{Maximum Resolution with a Telescope}
    	\begin{columns}[c]
    		\column{0.5\textwidth}
    		\begin{itemize}
    			\item Resolution is limited by diffraction: $ \theta \sim \frac{\lambda}{D}$
    	    	\item $\lambda$ : wavelength of light and $D$ : diameter of the telescope
    	    	\item Larger aperture $\Rightarrow$ better resolution
    	    	\item Even largest telescopes Gran Telescopio Canarias (GTC) $\Rightarrow$ limited to 13.3 milli-arcsecond scale
    	    	\item 
    		\end{itemize}

    \begin{frame}{Interferometry: The Technique}
    	\begin{columns}[c]
    		\column{0.65\textwidth}
    		\begin{itemize}
    			\item Two telescopes separated by a distance $B$ act like a single aperture of size $B$.
    			\item Angular resolution: $\theta \sim \frac{\lambda}{B}$
    			\item Interferometry measures correlation of signals from telescopes.
    			\item The measured quantity is the \textbf{visibility function}.
    			\item Visibility is related to the \textbf{Fourier transform of the source intensity}.
    			\item Power spectrum: $P(u,v) = |V(u,v)|^2$
    		\end{itemize}
